分频器 ​

根据情况改COUNT_MAX即可，这里是50mhz分2hz

verilog
module div_clk (
    input      sys_clk,   
    input      sys_rst_n,  
    output reg clk   
);

    // 计数器最大值： (50,000,000 / 2) / 2 - 1 = 12,499,999
    parameter COUNT_MAX = 25'd12_499_999; 
    reg [24:0] count; 

    always @(posedge sys_clk or negedge sys_rst_n) begin
        if (!sys_rst_n) begin
            count <= 25'd0;
            clk <= 1'b0;
        end else begin
            if (count == COUNT_MAX) begin
                count <= 25'd0;
                clk <= ~clk_;
            end else begin
                count <= count + 25'd1;
            end
        end
    end

endmodule
